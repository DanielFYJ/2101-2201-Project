@startuml SendStatus

actor RoboticCar

participant "//<<control>>//\n**RoboticCarManagement**" as A
participant "//<<control>>//\n**MovementManagement**" as B
participant "//<<control>>//\n**CommunicationManagement**" as C
participant "//<<control>>//\n**WebPortalManagement**" as D
participant "//<<entity>>// \n **WebPortal**" as E


RoboticCar -> A: Retrieve Data
activate A

A -> B: gatherStatus()
activate B

B --> A: success()
deactivate B

A -> C: establishCommunication()
activate C

C --> A: success()

C->A: receiveFeedback()
deactivate C

A->D: sendStatus()
activate D 

D-->A: success()
deactivate D

D->E: receiveStatus()
activate E

E --> A: Status received successfully
deactivate E
deactivate A


alt Robotic Car is unable to retrieve data from sensors
RoboticCar -> A: Retrieve Data

activate A
A -> B: gatherStatus()
activate B

B --> A: failure()
deactivate B

A -> C: establishCommunication()
activate C

C --> A: success()

C->A: receiveFeedback() (error feedback)
deactivate C

A -> D: sendFeedback()
activate D

D-->A: success()
deactivate D

D -> E: receiveFeedback()
activate E

E --> A: Feedback received successfully
deactivate E
deactivate A
end

alt Connection error between web portal and robotic car
RoboticCar -> A: Retrieve Data
activate A

A -> B: gatherStatus()
activate B

B --> A: success()
deactivate B

A -> C: establishCommunication()
activate C

C --> A: failure()
deactivate C

A -> E: Display error message 
activate E

E --> A: success()
deactivate E
deactivate A
end
@enduml
